    **Implement routes (GREEN phase):**

    Create `backend/src/routes/auth.ts`:
    ```typescript
    import { Router } from 'express';
    import bcrypt from 'bcryptjs';
    import { loginRateLimiter, loginSlowDown } from '../middleware/rateLimiter';

    export const authRouter = Router();

    authRouter.post('/login', loginSlowDown, loginRateLimiter, async (req, res) => {
      const { password } = req.body as { password: string };
      const hash = process.env.APP_PASSWORD_HASH;

      if (!hash || !password) {
        return res.status(401).json({ error: 'Invalid credentials' });
      }

      const match = await bcrypt.compare(password, hash);
      if (!match) {
        return res.status(401).json({ error: 'Invalid credentials' });
      }

      // Session fixation prevention (T-01-02)
      req.session.regenerate((err) => {
        if (err) return res.status(500).json({ error: 'Session error' });
        req.session.authenticated = true;
        req.session.save((err2) => {
          if (err2) return res.status(500).json({ error: 'Session save error' });
          res.json({ ok: true });
        });
      });
    });

    authRouter.get('/check', (req, res) => {
      if (req.session?.authenticated === true) {
        return res.json({ authenticated: true });
      }
      res.status(401).json({ authenticated: false });
    });

    authRouter.post('/logout', (req, res) => {
      req.session.destroy(() => {
        res.clearCookie('connect.sid');
        res.json({ ok: true });
      });
    });
    ```

    Mount in server.ts: add `app.use('/api/auth', authRouter)` BEFORE the requireAuth middleware (auth routes are public).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → POST /api/auth/login | Password sent in request body; bcrypt hash is in env var (never in DB or response) |
| Session cookie → subsequent requests | httpOnly cookie; never readable by JS |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Spoofing | POST /api/auth/login | mitigate | loginRateLimiter (5/15min/IP) + loginSlowDown applied before handler; generic error message prevents user enumeration |
| T-05-02 | Elevation of Privilege | Session fixation | mitigate | `req.session.regenerate()` called on every successful login — new session ID issued, old ID invalidated |
| T-05-03 | Info Disclosure | Login error messages | mitigate | Both "wrong password" and "missing password" return identical `{ error: 'Invalid credentials' }` — no oracle |
| T-05-04 | Tampering | APP_PASSWORD_HASH in env | accept | Changing the hash requires Render env var update + redeploy; acceptable for internal tool with small team |
</threat_model>
