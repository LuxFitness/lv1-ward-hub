Vacancy detection query (Pattern 2 — EXACT SQL from RESEARCH.md):
```sql
SELECT
  p.id            AS position_id,
  p.name          AS position_name,
  p.organization,
  p.sort_order,
  p.notes         AS consideration_notes,
  c.id            AS calling_id,
  c.member_id,
  m.name          AS member_name,
  c.status        AS calling_status,
  c.state_entered_at
FROM positions p
LEFT JOIN callings c
  ON c.position_id = p.id
  AND c.status NOT IN ('declined', 'released', 'cancelled')
LEFT JOIN members m
  ON m.id = c.member_id
WHERE p.is_active = true
ORDER BY p.sort_order, p.name;
```
NOTE: The LEFT JOIN uses `status NOT IN (...)` — not `status = 'set_apart'`. In-pipeline callings show their member too. Only rows where calling_id IS NULL are vacant.

Pending inbox query (Pattern 4 — EXACT SQL):
```sql
SELECT
  c.id,
  c.status,
  c.state_entered_at,
  EXTRACT(DAY FROM now() - c.state_entered_at) AS days_in_stage,
  p.name AS position_name,
  p.organization,
  m.name AS member_name
FROM callings c
JOIN positions p ON p.id = c.position_id
LEFT JOIN members m ON m.id = c.member_id
WHERE c.status NOT IN ('set_apart', 'declined', 'released', 'cancelled')
  AND (
    (c.status = 'recommended' AND c.state_entered_at < now() - INTERVAL '7 days')
    OR (c.status = 'extended'    AND c.state_entered_at < now() - INTERVAL '3 days')
    OR (c.status = 'accepted'    AND c.state_entered_at < now() - INTERVAL '14 days')
    OR (c.status = 'sustained'   AND c.state_entered_at < now() - INTERVAL '14 days')
  )
ORDER BY days_in_stage DESC;
```

    ```typescript
    import { supabase } from '../db';

    export const callingsRouter = Router();

    // CALL-01, CALL-02: Roster with vacancy detection via JOIN (D-10)
    callingsRouter.get('/', async (req, res) => {

      // Use: const { data, error } = await supabase.from('positions').select(`

      // Filter out declined/released/cancelled in application layer since Supabase JS
      // can't express the partial JOIN condition in select syntax.
      // Return flattened array; frontend groups by org.

      res.json(data);
    });

    // CALL-03: Create pipeline entry
    callingsRouter.post('/', async (req, res) => {
      const { position_id, member_id, bishopric_owner, notes } = req.body;
      const { data: calling, error } = await supabase
        .from('callings')
        .insert({ position_id, member_id, bishopric_owner, notes, status: 'recommended' })
        .select()
        .single();
      if (error) {
        // Unique index violation (CALL-07)
        if (error.code === '23505') {
          return res.status(409).json({ error: 'An active pipeline entry already exists for this position' });
        }
        return res.status(500).json({ error: error.message });
      }
      // D-11: Append calling_event
      await supabase.from('calling_events').insert({
        calling_id: calling.id,
        from_status: null,
        to_status: 'recommended',
      });
      res.status(201).json(calling);
    });

    // CALL-04, CALL-05, CALL-08: Advance pipeline stage
    callingsRouter.patch('/:id/transition', async (req, res) => {
      const { id } = req.params;
      const { status: toStatus, note } = req.body as { status: CallingStatus; note?: string };

      const { data: current, error: fetchError } = await supabase
        .from('callings').select('status').eq('id', id).single();
      if (fetchError || !current) return res.status(404).json({ error: 'Calling not found' });

      try {
        validateTransition(current.status as CallingStatus, toStatus);
      } catch (e) {
        return res.status(400).json({ error: (e as Error).message });
      }

      // D-13: MUST update state_entered_at on every transition
      const { data: updated, error: updateError } = await supabase
        .from('callings')
        .update({ status: toStatus, state_entered_at: new Date().toISOString(), updated_at: new Date().toISOString() })
        .eq('id', id)
        .select()
        .single();
      if (updateError) return res.status(500).json({ error: updateError.message });

      // D-11: Append calling_event audit record
      await supabase.from('calling_events').insert({
        calling_id: id,
        from_status: current.status,
        to_status: toStatus,
        note: note ?? null,
      });

      res.json(updated);
    });

    // CALL-06: Pending inbox
    callingsRouter.get('/pending', async (req, res) => {
      // Use Supabase RPC or raw SQL via supabase.rpc for the complex WHERE with intervals
      // Simplest pattern: query all in-progress callings and filter in JS
      const { data, error } = await supabase
        .from('callings')
        .select(`id, status, state_entered_at, positions(name, org_units(name)), members(name)`)
        .not('status', 'in', '("set_apart","declined","released","cancelled")');
      if (error) return res.status(500).json({ error: error.message });

      const thresholds: Record<string, number> = {
        recommended: 7, extended: 3, accepted: 14, sustained: 14,
      };
      const now = Date.now();
      const pending = (data ?? []).filter((c: any) => {
        const days = (now - new Date(c.state_entered_at).getTime()) / 86400000;
        return days >= (thresholds[c.status] ?? Infinity);
      }).map((c: any) => ({
        ...c,
        days_in_stage: Math.floor((now - new Date(c.state_entered_at).getTime()) / 86400000),
      }));

      res.json(pending.sort((a: any, b: any) => b.days_in_stage - a.days_in_stage));
    });
    ```
